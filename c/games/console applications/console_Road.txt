void gotoxy(int x, int y){

}

void premik(sme *me,char dir){
    if(dir=='a'){
        me->x-=2;
    }
    if(dir=='d'){
        me->x+=2;
    }
    if(dir=='w'){

        me->y-=2;


    }
    if(dir=='s'){
        me->y+=2;
    }


}
