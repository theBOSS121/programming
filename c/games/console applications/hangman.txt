#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(){
    srand(time(NULL));
char guessWords[][16] = {
    "luka",
    "telefon",
    "kemik",
    "aparat",
    "naprava",
    "krema",
};
int randomIndex = rand() % 6;
int numLives = 8;
int numCorrect=0;
int oldCorrect=0;
int lengthOfWord = strlen(guessWords[randomIndex]);
int letterGuessed[8] = {0,0,0,0,0,0,0,0};
int quit=0;
int i;

char guess[16];
char letterEntered;


while(numCorrect < lengthOfWord){
       printf("hangman word:");

        for(i=0;i<lengthOfWord;i++){
        if(letterGuessed[i]==1){
            printf("%c",guessWords[randomIndex][i]);
        }else{
            printf("-");
        }

        }
        printf("\n");
    printf("correct so far %d\n",numCorrect);
    printf("enter a guess letter:");
    fgets(guess,16,stdin);
    if(strncmp(guess,"quit", 4)==0){
        quit=1;
        break;
    }
    letterEntered = guess[0];
    printf("letterEntered:%c\n",letterEntered);

    oldCorrect=numCorrect;
    for(i=0;i<lengthOfWord;i++){
        if(letterGuessed[i]==1){
            continue;
        }
        if(letterEntered==guessWords[randomIndex][i]){
            letterGuessed[i] = 1;
            numCorrect++;

        }
    }
    if(oldCorrect==numCorrect){
        numLives--;
        printf("sorry wrong guess\n\n\n");
        if(numLives==0){
            break;
        }
    }else{
        printf("correct guess :)\n\n\n");
    }

}
if(quit==1){
    printf("\nthe user quit early\n");
}else if(numLives==0){
printf("\nsorry you lose, the word was %s\n",guessWords[randomIndex]);
}else{
printf("\nyou win!!! :)\n");
}


return 0;
}
